Controller
  angular.module('app').controller('PresentationCtrl',function($scope,$location,$routeParams){

    $scope.pageNr = $routeParams.nr || 1;

    $scope.$watch('pageNr',function(value,old){
        $location.path('/slides/'+value);
    });

    $scope.next = function(){
      $scope.pageNr++;
    };

    $scope.prev = function(){
      $scope.pageNr--;
    };

  });
