Directive
  <pre figlet="doh">awesome</pre>

  angular.module('app').directive('figlet',function(figlet){
    return {
      scope: false,
      link: function(scope,element,attrs){

        attrs.$observe('figlet',function(font){
          
          if (font) {
            figlet(element.text(),font).then(function(txt){
              element.text(txt);
            });
          }  
          
        });
      
      }
    };
  });
